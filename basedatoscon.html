Reemplaza el host y la contraseña con los valores reales de tu Nodo Remoto antes de ejecutar.
Conexión Multi-Nodo Connection Pooler :6543 
-- 1. Habilitar extensión FDW
CREATE EXTENSION IF NOT EXISTS postgres_fdw;

-- 2. Registrar servidor foráneo (Nodo Remoto)
DROP SERVER IF EXISTS servidor_sucursal CASCADE;
CREATE SERVER servidor_sucursal
  FOREIGN DATA WRAPPER postgres_fdw
  OPTIONS (host 'db.XXXXXXXXXXXX.supabase.co', dbname 'postgres', port '6543');

-- 3. Mapeo de credenciales
CREATE USER MAPPING FOR postgres
  SERVER servidor_sucursal
  OPTIONS (user 'postgres', password 'TU_CONTRASEÑA');

-- Vista Global Unificada (abstrae el origen físico del dato)
CREATE OR REPLACE VIEW public.vista_global_estudiantes AS
SELECT matricula, persona_id, carrera, semestre, estado, nombre,
       'Local (Nodo Central)'::text AS origen_datos
FROM public.estudiantes WHERE estado = 'Activo'
UNION ALL
SELECT matricula, persona_id, carrera, semestre, estado, nombre,
       'Remoto (Nodo Sucursal)'::text AS origen_datos
FROM de_paso.estudiantes_remotos_link;

-- Transacción 1: Escritura fragmentada vertical
CREATE OR REPLACE FUNCTION registrar_persona_distribuida(
  p_nombre text, p_apellido text, p_rol_id integer,
  p_foto_url text, p_foto_encoding text
) RETURNS integer LANGUAGE plpgsql AS $$
DECLARE nuevo_id integer;
BEGIN
  INSERT INTO public.personas (nombre, apellido, rol_id, activo)
  VALUES (p_nombre, p_apellido, p_rol_id, true)
  RETURNING id INTO nuevo_id;

  INSERT INTO de_paso.personas_biometria_link (id, foto_url, foto_encoding)
  VALUES (nuevo_id, p_foto_url, p_foto_encoding);

  RETURN nuevo_id;
END; $$;

-- Transacción 2: Logs de accesos + presencia en tiempo real
CREATE OR REPLACE FUNCTION registrar_acceso_y_presencia(
  p_persona_id integer, p_matricula text,
  p_carrera text, p_tipo text, p_camara text
) RETURNS void LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO public.entradas_salidas (persona_id, matricula, carrera, tipo, camara)
  VALUES (p_persona_id, p_matricula, p_carrera, p_tipo, p_camara);

  INSERT INTO public.presencia_actual (persona_id, dentro, ultima_actualizacion)
  VALUES (p_persona_id, (p_tipo = 'Entrada'), now())
  ON CONFLICT (persona_id)
  DO UPDATE SET dentro = (p_tipo = 'Entrada'), ultima_actualizacion = now();
END; $$;

-- Transacción 3: Control de emergencias con cascada
CREATE OR REPLACE FUNCTION activar_bloqueo_emergencia(
  p_codigo_id uuid, p_tipo text
) RETURNS void LANGUAGE plpgsql AS $$
BEGIN
  INSERT INTO public.emergencias_activas (codigo_id, tipo, activo)
  VALUES (p_codigo_id, p_tipo, true);

  UPDATE public.modo_acceso
  SET modo = 'bloqueo_total', actualizado_en = now()
  WHERE id = 1;
END; $$;

CREATE OR REPLACE FUNCTION obtener_dashboard_distribuido()
RETURNS TABLE (
  matricula  text,
  nombre     character varying,
  carrera    text,
  estado     text,
  nodo       text
) LANGUAGE plpgsql AS $$
BEGIN
  RETURN QUERY
  SELECT v.matricula, v.nombre, v.carrera, v.estado, v.origen_datos
  FROM public.vista_global_estudiantes v
  ORDER BY v.carrera, v.matricula;
END; $$;
